#include<iostream>
#include<algorithm>
using namespace std;
int a[500005];
int main()
{
	int n,i,j,mid;
	int count=0; 
	cin>>n;
	mid=n/2-1;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	i=mid;
	j=n-1;
	while(i>=0)
	{
		if(a[i]*2<=a[j])
		{
			count++;
			i--;
			j--;
		}
		else
		{
			i--;
		}
	}
	cout<<n-count;
}
