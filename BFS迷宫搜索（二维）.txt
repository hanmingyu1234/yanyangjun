#include <iostream>
#include <queue>
using namespace std;
struct point
{
	int x;
	int y;
	int step;
};
queue<point>r;
int a[100][100];
int b[100][100];
int dx[4]={0,1,0,-1};
int dy[4]={1,0,-1,0};
int main()
{
	int n,m,i,j;
	cin>>n>>m;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	point start;
	start.x=0;
	start.y=0;
	start.step=0;
	r.push(start);
	b[0][0]=1;
	while(!r.empty())
	{
		int x=r.front().x,y=r.front().y;
		if(x==n-1&&y==m-1)
		{
			printf("%d",r.front().step);
			break;
		}
		for(int k=0;k<4;k++)
		{
			int tx,ty;
			tx=x+dx[k];
			ty=y+dy[k];
			if(a[tx][ty]==0&&b[tx][ty]==0&&tx>=0&&ty>=0&&tx<n&&ty<m)
			{
				point temp;
				temp.x=tx;
				temp.y=ty;
				temp.step=r.front().step+1;
				r.push(temp);
				b[tx][ty]=1;
			}
		}
		r.pop();
	}
}
