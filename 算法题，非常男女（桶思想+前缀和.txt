#include <iostream>
using namespace std;
int a[1000005];
int b[1000005];
int c[1000005];
int main()
{
	int n,i;
	int x=0; 
	int ans=0;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		if(a[i]==0)
		{
			a[i]=-1;
		}
		x=x+a[i];
		b[i]=x;
	}
	for(i=1;i<=n;i++)
	{
		if(c[b[i]+n]==0)
		{
			c[b[i]+n]=i;
		}
	}
	for(i=1;i<=n;i++)
	{
		if(b[i]==0)
		{
			ans=max(ans,i);
			continue;
		}
		ans=max(ans,i-c[b[i]+n]);
	}	
	cout<<ans<<endl;
}
