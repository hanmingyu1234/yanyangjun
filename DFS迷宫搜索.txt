#include<iostream>
#include<algorithm>
using namespace std;
int min(int a,int b)
{
	if(a<b)
	{
		return a;
	}
	else
	{
		return b;
	}
}
int main()
{
	int n,m,i,j;
	int count=0; 
	int a[1005];
	int b[1005];
	int fact=0;
	int fact1=0;
	int fact2=0;
	int num=0;
	cin>>n;
	for(j=0;j<n;j++)
	{
		count=0;
		fact=0;
		num=0;
		fact1=0;
		fact2=0;
		cin>>m;
		for(i=0;i<m;i++)
		{
			scanf("%d",&a[i]);
			count++;
		}
		sort(a,a+count);
		while(count>=4)
		{
			fact1=a[1]+a[0]+a[count-1]+a[1];
			fact2=a[count-1]+a[0]+a[count-2]+a[0];
			fact=fact+min(fact1,fact2);
			count=count-2;
		}
		if(count==0)
			fact=fact+0;
		else if(count==1)
			fact=fact+a[0];
		else if(count==2)
			fact=fact+a[1];
		else if(count==3)
			fact=fact+a[0]+a[1]+a[2];
//		else if(count==4)
//			fact=fact+a[1]+a[0]+a[3]+a[1]+a[1];
		b[j]=fact;
	}
	for(j=0;j<n;j++)
	{
		printf("%d\n",b[j]);
	}
}
