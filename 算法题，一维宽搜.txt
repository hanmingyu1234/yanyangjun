#include <iostream>
#include <queue>
using namespace std;
struct point
{
	int x;
	int step;
};
queue<point>r;
int b[1000005];
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	if(n>=m)
	{
		printf("%d",n-m);
		return 0;
	}
	point start;
	start.x=n;
	start.step=0;
	r.push(start);
	b[n]=1;
	while(!r.empty())
	{
		int x=r.front().x;
		if(x==m)
		{
			printf("%d",r.front().step);
			break;
		}
		if(b[x-1]==0&&x>0)
		{
			point temp1;
			temp1.x=x-1;
			temp1.step=r.front().step+1;
			r.push(temp1);
			b[x-1]=1;
		} 
		if(b[x+1]==0&&x<m)
		{
			point temp2;
			temp2.x=x+1;
			temp2.step=r.front().step+1;
			r.push(temp2);
			b[x+1]=1;
		}
		if(b[x*2]==0&&x<m)
		{
			point temp3;
			temp3.x=x*2;
			temp3.step=r.front().step+1;
			r.push(temp3);
			b[x*2]=1;
		}
		r.pop();
	}
}
